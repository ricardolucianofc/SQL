SELECT

(LEFT(NOME_COMPLETO, CASE WHEN CHARINDEX(' ',NOME_COMPLETO) = 0 THEN LEN(NOME_COMPLETO) ELSE CHARINDEX(' ',NOME_COMPLETO) -1 END)) AS NOME,

(SELECT RIGHT(NOME_COMPLETO, CASE WHEN CHARINDEX(' ',REVERSE(NOME_COMPLETO)) = 0 THEN 0 ELSE CHARINDEX(' ',REVERSE(NOME_COMPLETO))-1 END)) AS SOBRENOME

FROM INTERDOC..PESSOAS

WHERE NOME IN ('AEM','RACR','RLFC','BS')
